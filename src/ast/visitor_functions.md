- visitDeclarator
- visitDeclaration
- visitEnum
- visitExpr

- visitBExpr

- visitBArithExpr
- visitBLogicExpr
- visitBCompExpr
- visitSubscriptExpr
- visitMemberAccessExpr
- visitPtrMemberAccessExpr
- visitCExpr
- visitFncCallExpr
- visitCastExpr
- visitCondExpr
- visitFnc
- visitIdentifier
- visitTagname
- visitPointer
- visitCStmt
- visitEStmt
- visitSStmt
- visitIStmt
- visitJStmt
- visitReturn
- visitStruct
- visitUnion
- visitTypedef
